# jQuery的初步构建 #

```
var $ = jQuery = function() {
    return jQuery.fn.init();
};
jQuery.fn = jQuery.prototype = {
    init: function() {
        this.length = 0;
        this.test = function() {
            return this.length;
        }
        return this;
    },
    jquery: '1.3.2',
    length: 1,
    size: function() {
        return this.length;
    }
};
jQuery.fn.init.prototype = jQuery.fn; //jQuery的原型对象覆盖了init的原型对象


console.log($().jquery); // '1.3.2'
console.log($().test()); // 0
console.log($().size()); // 0
```
