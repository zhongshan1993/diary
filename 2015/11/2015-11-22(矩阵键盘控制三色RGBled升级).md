# 矩阵键盘控制三色RGBled发光升级 #

**今天加上了自动循环播放的功能**

```
#include<reg52.h>
/*RGB灯*/
sbit red = P1^0;
sbit green = P1^1;
sbit blue = P1^2;

/*蜂鸣器、数码管的段和位*/
sbit beep = P2^3;
sbit dula = P2^6;
sbit wela = P2^7;

/*独立键盘K1*/
sbit K1 = P3^4;

#define uchar unsigned char
#define uint unsigned int
#define PERIOD 257
#define HighInitValue 0xFF
#define LowInitValue 0xEC
uint t0, t1;
uint up, down;
uint redValue, greenValue, blueValue;  // 红绿蓝颜色值

uchar code table[] = {0x3f, 0x06, 0x5b, 0x4f, 0x66, 0x6d, 0x7d, 0x07, 0x7f, 0x6f, 0x77, 0x7c, 0x39, 0x5e, 0x79, 0x71}; //0-F数码管显示代码
uchar codeTemp, codeKey; // 用于判断矩阵键盘
uchar number[6]; // 数码管显示序列
uchar numberCount = 0; // 计算数码管显示序列已经参存入的位数
uchar numT, numI; // 用于数码管延时显示的计数
uchar code local[6] = {0xfe, 0xfd, 0xfb, 0xf7, 0xef, 0xdf}; // 数码管显示位置的数组
void Delay10ms()
{
    unsigned char i, j;
    i = 10;
    j = 10;
    do{
        while (--j);
    } while (--i);
}
/*将自动调节的三色RGB值转化为number数组,在数码管中显示*/
void RGBConvertToNumber(uint numberR, uint numberG, uint numberB) {
    number[0] = numberR/16;
    number[1] = numberR%16;
    
    number[2] = numberG/16;
    number[3] = numberG%16;
    
    number[4] = numberB/16;
    number[5] = numberB%16; 

    numberCount = 6; // 触发数码管显示
}
/*键盘扫描*/
void keyBoardScan() {
    P3=0xfe;
    codeTemp=P3;
    codeTemp=codeTemp&0xf0;
    if(codeTemp != 0xf0)
    {
        Delay10ms();
        if(codeTemp!=0xf0)
        {
            codeTemp=P3;
            switch(codeTemp)
            {
                case 0xee:
                   codeKey=0;
                   break;
                case 0xde:
                    codeKey=1;
                    break;
                case 0xbe:
                    codeKey=2;
                    break;
                case 0x7e:
                    codeKey=3;
                    break;
            }
            while(codeTemp!=0xf0)
            {
                codeTemp=P3;
                codeTemp=codeTemp&0xf0;
                beep=0;
            }
            beep=1;
            if (numberCount < 6) {
                number[numberCount] = codeKey;
                numberCount++;
            }
        }
    }
    P3=0xfd;
    codeTemp=P3;
    codeTemp=codeTemp&0xf0;
    if(codeTemp!=0xf0)
    {
        Delay10ms();
        if(codeTemp!=0xf0)
        {
            codeTemp=P3;
            switch(codeTemp)
            {
                case 0xed:
                    codeKey=4;
                    break;
                case 0xdd:
                    codeKey=5;
                    break;
                case 0xbd:
                    codeKey=6;
                    break;
                case 0x7d:
                    codeKey=7;
                    break;
            }
            while(codeTemp!=0xf0)
            {
                codeTemp=P3;
                codeTemp=codeTemp&0xf0;
                beep=0;
            }
            beep=1;
            if (numberCount < 6) {
                number[numberCount] = codeKey;
                numberCount++;
            }
        }
    }
    P3=0xfb;
    codeTemp=P3;
    codeTemp=codeTemp&0xf0;
    if(codeTemp!=0xf0)
    {
        Delay10ms();
        if(codeTemp!=0xf0)
        {
            codeTemp=P3;
            switch(codeTemp)
            {
                case 0xeb:
                    codeKey=8;
                    break;
                case 0xdb:
                    codeKey=9;
                    break;
                case 0xbb:
                    codeKey=10;
                    break;
                case 0x7b:
                    codeKey=11;
                    break;
            }
            while(codeTemp!=0xf0)
            {
                codeTemp=P3;
                codeTemp=codeTemp&0xf0;
                beep=0;
            }
            beep=1;
            if (numberCount < 6) {
                number[numberCount] = codeKey;
                numberCount++;
            }
        }
    }
    P3=0xf7;
    codeTemp=P3;
    codeTemp=codeTemp&0xf0;
    if(codeTemp!=0xf0)
    {
        Delay10ms();
        if(codeTemp!=0xf0)
        {
            codeTemp=P3;
            switch(codeTemp)
            {
                case 0xe7:
                    codeKey=12;
                    break;
                case 0xd7:
                    codeKey=13;
                    break;
                case 0xb7:
                    codeKey=14;
                    break;
                case 0x77:
                    codeKey=15;
                    break;
            }
            while(codeTemp!=0xf0)
            {
                codeTemp=P3;
                codeTemp=codeTemp&0xf0;
                beep=0;
            }
            beep=1;
            if (numberCount < 6) {
                number[numberCount] = codeKey;
                numberCount++;
            }
        }
    }
    
}

void main(){
    uint numberR = 255, numberG = 0, numberB = 0;
    int step = 1;
    uchar startColor = 2;
    /*定时器初始化*/
    TMOD = 0x11;
    TH0 = HighInitValue; TL0 = LowInitValue;
    TH1 = HighInitValue; TL1 = LowInitValue;
    EA = 1;
    ET0 = 1; ET1 = 1;
    TR0 = 1; TR1 = 1;

    t1 = 0; t0 = 0; numT = 0; numI = 0; 
 
    while(1) {
        dula=0;
        wela=0;
        
        if ( 1 ) {
            if (startColor == 3) {
                startColor = 0;
            } else if (startColor == 2) {
                do {
                    numberB += step;
                    
                    redValue = PERIOD - 1 - numberR;
                    greenValue = PERIOD - 1 - numberG;
                    blueValue = PERIOD - 1 - numberB;

                    RGBConvertToNumber(numberR, numberG, numberB);

                    Delay10ms();
                } while(numberB>=1 && numberB<=254);
                step = -step;
                startColor++;
                if(startColor == 3) startColor = 0;
            } else if (startColor == 0) {
                do {
                    numberR += step;
                    
                    redValue = PERIOD - 1 - numberR;
                    greenValue = PERIOD - 1 - numberG;
                    blueValue = PERIOD - 1 - numberB;

                    RGBConvertToNumber(numberR, numberG, numberB);

                    Delay10ms();
                } while(numberR>=1 && numberR<=254);
                step = -step;
                startColor++;
                if(startColor == 3) startColor = 0;
            } else if (startColor == 1) {
                do {
                    numberG += step;
                    
                    redValue = PERIOD - 1 - numberR;
                    greenValue = PERIOD - 1 - numberG;
                    blueValue = PERIOD - 1 - numberB;

                    RGBConvertToNumber(numberR, numberG, numberB);

                    Delay10ms();
                } while(numberG>=1 && numberG<=254);
                step = -step;
                startColor++;
                if(startColor == 3) startColor = 0;
            } 
        } else {
            keyBoardScan();

            if (numberCount == 6) {
                redValue = PERIOD - 1 - (number[0]*16 + number[1]);
                greenValue = PERIOD - 1 - (number[2]*16 + number[3]);
                blueValue = PERIOD - 1 - (number[4]*16 + number[5]);
            }
        }
    
    }
}
/*定时器T1*/
void time1_int(void) interrupt 3 using 1
{
    t1++;
    TH1 = HighInitValue; TL1 = LowInitValue; //设置初值

    //红灯亮
    if (t1 == redValue) {
        red = 0;
    }
    if (t1 == greenValue) {
        green = 0;
    }
    if (t1 == blueValue) {
        blue = 0;
    }
}
/*定时器T0*/
void time0_int(void) interrupt 1 using 1  
{
    t0++;
    numT++;

    TH0 = HighInitValue; TL0 = LowInitValue; //设置初值
    if (t0 == PERIOD) {

        /* 灯灭 */
        red = 1; 
        green = 1;
        blue = 1;

        t0 = 0;
        t1 = 0;
    }
    
    if (numT == 30) {
        if (numberCount > 0) {
 
            P0 = table[number[numI]];
            dula = 1;
            dula = 0;
            P0 = 0xff; // 全部重置
            P0 = local[numI]; // 显示的位置
            wela = 1;
            wela = 0;       
    
            numT = 0;
            if (numI < numberCount-1) {
                numI++; 
            } else {
                numI = 0;
            } 
            
        }  
    }
}
``` 
